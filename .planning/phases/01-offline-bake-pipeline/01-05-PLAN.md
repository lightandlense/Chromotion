```python
"""
Motion Review Tool — Tkinter UI for visual QA of SAM 2 tracking output.

Displays:
- Per-part per-frame mask overlays on animation frames
- Flagged frames highlighted (orange = interpolated, red = drift block)
- Part selector, frame slider, frame info panel

Usage:
    python src/offline/motion_review_tool.py \\
        --motion-data data/motion_data.json \\
        --masks data/rest_pose_masks \\
        --frames src/animations/ram_frames

Keyboard shortcuts:
    Left/Right arrow: prev/next frame
    Up/Down arrow: prev/next part
    Q: quit

Note: Brush correction is deferred (see CONTEXT.md). This is view-only for Phase 1.
"""
import argparse
import json
import pathlib
import tkinter as tk
from tkinter import ttk, messagebox
from PIL import Image, ImageTk, ImageDraw
import numpy as np


class MotionReviewTool:
    CANVAS_W = 960   # Display width (half of 1920)
    CANVAS_H = 540   # Display height (half of 1080)
    OVERLAY_ALPHA = 120  # Mask overlay opacity (0-255)

    def __init__(self, root: tk.Tk, motion_data: dict, masks_dir: str, frames_dir: str):
        self.root = root
        self.motion_data = motion_data
        self.masks_dir = pathlib.Path(masks_dir)
        self.frames_dir = pathlib.Path(frames_dir)

        self.parts = list(motion_data["parts"].keys())
        self.frame_count = motion_data["frame_count"]
        self.rest_pose_frame = motion_data.get("rest_pose_frame", 0)

        self.current_part_idx = 0
        self.current_frame = 0

        # Precompute flagged frames per part
        self.flagged = self._compute_flagged()

        # Load rest pose masks (one per part, static)
        self.rest_masks = {}
        for part in self.parts:
            mask_path = self.masks_dir / f"{part}.png"
            if mask_path.exists():
                self.rest_masks[part] = Image.open(mask_path).convert("RGBA")

        self._build_ui()
        self._update_display()

    def _compute_flagged(self) -> dict:
        """
        Return {part_name: set(frame_indices)} for flagged frames.
        Flagged = interpolated:true OR in a drift_block.
        """
        flagged = {}
        for part_name, part_data in self.motion_data["parts"].items():
            flagged_set = set()
            for frame in part_data["frames"]:
                if frame.get("interpolated", False):
                    flagged_set.add(frame["frame"])
            for block in part_data.get("drift_blocks", []):
                for i in range(block["start_frame"], block["end_frame"] + 1):
                    flagged_set.add(i)
            flagged[part_name] = flagged_set
        return flagged

    def _build_ui(self):
        self.root.title("Motion Review Tool — Color Animals Interactive")
        self.root.configure(bg="#1e1e1e")

        # Top: part selector
        top_frame = tk.Frame(self.root, bg="#1e1e1e")
        top_frame.pack(side=tk.TOP, fill=tk.X, padx=8, pady=4)

        tk.Label(top_frame, text="Part:", fg="white", bg="#1e1e1e").pack(side=tk.LEFT)
        self.part_var = tk.StringVar(value=self.parts[0])
        part_menu = ttk.Combobox(top_frame, textvariable=self.part_var, values=self.parts, width=16)
        part_menu.pack(side=tk.LEFT, padx=4)
        part_menu.bind("<<ComboboxSelected>>", self._on_part_change)

        # Canvas
        self.canvas = tk.Canvas(
            self.root, width=self.CANVAS_W, height=self.CANVAS_H,
            bg="black", highlightthickness=0
        )
        self.canvas.pack(padx=8, pady=4)

        # Frame slider
        slider_frame = tk.Frame(self.root, bg="#1e1e1e")
        slider_frame.pack(side=tk.TOP, fill=tk.X, padx=8)

        tk.Label(slider_frame, text="Frame:", fg="white", bg="#1e1e1e").pack(side=tk.LEFT)
        self.frame_var = tk.IntVar(value=0)
        self.slider = tk.Scale(
            slider_frame, from_=0, to=self.frame_count - 1,
            orient=tk.HORIZONTAL, variable=self.frame_var,
            command=self._on_frame_change,
            bg="#2d2d2d", fg="white", highlightthickness=0,
            troughcolor="#444", length=700
        )
        self.slider.pack(side=tk.LEFT, fill=tk.X, expand=True)

        # Info panel
        self.info_var = tk.StringVar()
        tk.Label(self.root, textvariable=self.info_var, fg="white", bg="#1e1e1e",
                 font=("Courier", 10), anchor="w", justify=tk.LEFT
                 ).pack(side=tk.TOP, fill=tk.X, padx=8, pady=2)

        # Legend
        legend = tk.Label(
            self.root,
            text="Legend: [GREEN] normal  [ORANGE] interpolated  [RED] drift block  |  Arrow keys: nav  Q: quit",
            fg="#888", bg="#1e1e1e", font=("Courier", 9)
        )
        legend.pack(side=tk.BOTTOM, fill=tk.X, padx=8, pady=2)

        # Key bindings
        self.root.bind("<Left>", lambda e: self._step_frame(-1))
        self.root.bind("<Right>", lambda e: self._step_frame(1))
        self.root.bind("<Up>", lambda e: self._step_part(-1))
        self.root.bind("<Down>", lambda e: self._step_part(1))
        self.root.bind("<q>", lambda e: self.root.quit())
        self.root.bind("<Q>", lambda e: self.root.quit())

    def _on_part_change(self, event=None):
        part_name = self.part_var.get()
        self.current_part_idx = self.parts.index(part_name)
        self._update_display()

    def _on_frame_change(self, value):
        self.current_frame = int(value)
        self._update_display()

    def _step_frame(self, delta: int):
        new_frame = max(0, min(self.frame_count - 1, self.current_frame + delta))
        self.current_frame = new_frame
        self.frame_var.set(new_frame)
        self._update_display()

    def _step_part(self, delta: int):
        new_idx = max(0, min(len(self.parts) - 1, self.current_part_idx + delta))
        self.current_part_idx = new_idx
        self.part_var.set(self.parts[new_idx])
        self._update_display()

    def _load_frame_image(self, frame_idx: int) -> Image.Image:
        """Load animation frame. SAM 2 frames are 1-indexed (0001.jpg = frame 0)."""
        frame_path = self.frames_dir / f"{frame_idx + 1:04d}.jpg"
        if frame_path.exists():
            return Image.open(frame_path).convert("RGBA")
        # Fallback: blank frame
        return Image.new("RGBA", (1920, 1080), (0, 0, 0, 255))

    def _get_frame_data(self, part_name: str, frame_idx: int) -> dict:
        frames = self.motion_data["parts"][part_name]["frames"]
        return next((f for f in frames if f["frame"] == frame_idx), {})

    def _update_display(self):
        part_name = self.parts[self.current_part_idx]
        frame_idx = self.current_frame
        flagged_set = self.flagged.get(part_name, set())

        # Load base animation frame
        base_img = self._load_frame_image(frame_idx)
        base_img = base_img.resize((self.CANVAS_W, self.CANVAS_H), Image.LANCZOS)

        # Overlay rest pose mask (tinted, showing where this part's texture will be sampled)
        if part_name in self.rest_masks:
            mask_img = self.rest_masks[part_name].resize(
                (self.CANVAS_W, self.CANVAS_H), Image.NEAREST
            )
            # Determine overlay color based on frame status
            if frame_idx in flagged_set:
                frame_data = self._get_frame_data(part_name, frame_idx)
                if frame_data.get("interpolated", False):
                    overlay_color = (255, 165, 0, self.OVERLAY_ALPHA)  # orange: interpolated
                else:
                    overlay_color = (255, 80, 80, self.OVERLAY_ALPHA)  # red: drift block
            else:
                overlay_color = (80, 255, 80, self.OVERLAY_ALPHA)  # green: normal

            # Apply colored overlay where mask alpha > 0
            colored_mask = Image.new("RGBA", mask_img.size, overlay_color)
            mask_alpha = mask_img.split()[3]
            base_img.paste(colored_mask, mask=mask_alpha)

        # Draw centroid crosshair
        frame_data = self._get_frame_data(part_name, frame_idx)
        if frame_data.get("cx") is not None:
            cx_scaled = frame_data["cx"] * self.CANVAS_W / 1920
            cy_scaled = frame_data["cy"] * self.CANVAS_H / 1080
            draw = ImageDraw.Draw(base_img)
            r = 8
            draw.line([(cx_scaled - r, cy_scaled), (cx_scaled + r, cy_scaled)], fill=(255, 255, 0, 255), width=2)
            draw.line([(cx_scaled, cy_scaled - r), (cx_scaled, cy_scaled + r)], fill=(255, 255, 0, 255), width=2)

        # Update canvas
        self.tk_img = ImageTk.PhotoImage(base_img)
        self.canvas.create_image(0, 0, anchor=tk.NW, image=self.tk_img)

        # Update info
        status = "FLAGGED" if frame_idx in flagged_set else "OK"
        q = frame_data.get("tracking_quality", "N/A")
        interp = frame_data.get("interpolated", False)
        info = (
            f"Part: {part_name} ({self.current_part_idx + 1}/{len(self.parts)})  "
            f"Frame: {frame_idx}/{self.frame_count - 1}  "
            f"Quality: {q:.3f}  "
            f"Interpolated: {interp}  "
            f"Status: {status}  "
            f"Flagged in part: {len(flagged_set)} frame(s)"
        )
        self.info_var.set(info)


def main():
    parser = argparse.ArgumentParser(description="Motion Review Tool — visual QA for SAM 2 tracking")
    parser.add_argument("--motion-data", default="data/motion_data.json", help="Path to motion_data.json")
    parser.add_argument("--masks", default="data/rest_pose_masks", help="Path to rest_pose_masks directory")
    parser.add_argument("--frames", default="src/animations/ram_frames", help="Path to JPEG frames directory")
    args = parser.parse_args()

    with open(args.motion_data) as f:
        motion_data = json.load(f)

    root = tk.Tk()
    app = MotionReviewTool(root, motion_data, args.masks, args.frames)
    root.mainloop()


if __name__ == "__main__":
    main()
```
